Background Size

background-size();

  • @include background-size(auto);
  • @include background-size(contain);
  • @include background-size(cover);
  • @include background-size(●px ●px); //例）10px 10px
  • @include background-size(●% auto);//例）50% auto

サンプル

background-size(auto);

background-size(contain);

background-size(cover);

background-size(10px,10px);

background-size(50%,auto);

Border Radius

border-radius();

  • @include border-radius($radius, $vertical-radius);
  • @include border-radius();//defautl = 5px
  • @include border-radius(10px);
  • @include border-radius(10px 20px);
  • @include border-radius(10px 20px 30px);
  • @include border-radius(10px 20px 30px 40px);
  • @include border-radius(10px 20px 30px 40px, 40px 30px 20px 10px);

サンプル

border-radius();

border-radius(40px);

border-radius(40px 20px);

border-radius(40px 20px 30px);

border-radius(40px 20px 30px 10px);

Box Shadow

box-shadow();

  • @include box-shadow($h-offset,$v-offset,$blur-length,$spread-length,$color,$inset);
  • @include box-shadow();//0px 0px 5px #333333;
  • @include box-shadow(20px 20px);
  • @include box-shadow(20px 20px 20px);
  • @include box-shadow(20px 20px 20px 10px);
  • @include box-shadow(20px 20px 20px 10px rgba(255,0,0,0.5));
  • @include box-shadow(20px 20px 20px 10px rgba(255,0,0,0.5) inset);

サンプル

box-shadow();//0px 0px 5px #333333;

box-shadow(20px 20px);

box-shadow(20px 20px 20px);

box-shadow(20px 20px 20px 10px);

box-shadow(20px 20px 20px 10px rgba(255,0,0,0.5));

box-shadow(20px 20px 20px 10px rgba(255,0,0,0.5) inset);
